{
	private int num;
	public MyNum(int num)
	{
		this.num=num;
	}
	public String toString()
	{
		return "["+num+"]";
	}
	public int compareTo(MyNum o)
	{
		if(this.num>o.num)
		{
			return 1;
		}
		else if(this.num<o.num)
		{
			return -1;
		}
		else
		{
			return 0;
		}
	}
}
